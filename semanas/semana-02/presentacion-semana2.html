Math.cos(0);                    // 1

// Ejemplo práctico: generar número aleatorio en rango
function numeroAleatorio(min, max) {
    return Math.floor(Math.random() * (max - min + 1)) + min;
}

// Ejemplo: calcular distancia entre puntos
function distancia(x1, y1, x2, y2) {
    const dx = x2 - x1;
    const dy = y2 - y1;
    return Math.sqrt(Math.pow(dx, 2) + Math.pow(dy, 2));
}

numeros.pop();                 // [1, 2, 3] - quitar del final
numeros.unshift(0);            // [0, 1, 2, 3] - agregar al inicio
numeros.shift();               // [1, 2, 3] - quitar del inicio

// Métodos de transformación (funcionales)
const duplicados = numeros.map(x => x * 2);        // [2, 4, 6]
const pares = numeros.filter(x => x % 2 === 0);    // [2]
const suma = numeros.reduce((acc, x) => acc + x, 0); // 6

numeros.indexOf(2);            // 1 - posición del elemento
numeros.find(x => x > 2);      // 3 - primer elemento que cumple
numeros.some(x => x > 5);      // false - alguno cumple
numeros.every(x => x > 0);     // true - todos cumplen

// Ejemplo práctico: validar y formatear email
function procesarEmail(email) {
    const emailLimpio = email.trim().toLowerCase();
    
    if (!emailLimpio.includes("@") || !emailLimpio.includes(".")) {
        return { valido: false, mensaje: "Email inválido" };
    }
    
    const [usuario, dominio] = emailLimpio.split("@");
    return {
        valido: true,
        usuario: usuario,
        dominio: dominio,
        emailFormateado: emailLimpio
    };
}

// Ejemplo: crear slug para URL
function crearSlug(titulo) {
    return titulo
        .toLowerCase()
        .trim()
        .replace(/[^\w\s-]/g, '') // quitar caracteres especiales
        .replace(/[\s_-]+/g, '-') // reemplazar espacios con guiones
        .replace(/^-+|-+$/g, ''); // quitar guiones del inicio/final
}
// "Mi Artículo Genial!" → "mi-articulo-genial"

Formatos según tiempo y proximidad

// Formato condicional según el momento del día
const formatoSaludo = (fecha) => {
    const hora = fecha.getHours();
    const saludo = hora < 12 ? 'Buenos días' : hora < 18 ? 'Buenas tardes' : 'Buenas noches';
    return format(fecha, `'${saludo}, hoy es' EEEE dd 'de' MMMM`);
};

// Formato según la proximidad de la fecha
const formatoProximidad = (fecha) => {
    const ahora = new Date();
    const diferenciaDias = differenceInDays(fecha, ahora);
    
    if (diferenciaDias === 0) {
        return format(fecha, "'Hoy a las' HH:mm");
    } else if (diferenciaDias === 1) {
        return format(fecha, "'Mañana a las' HH:mm");
    } else if (diferenciaDias === -1) {
        return format(fecha, "'Ayer a las' HH:mm");
    } else if (diferenciaDias > 0 && diferenciaDias <= 7) {
        return format(fecha, "EEEE 'a las' HH:mm");
    } else {
        return format(fecha, "dd/MM/yyyy 'a las' HH:mm");
    }
};

// Formato según el estado del evento
const formatoEstadoEvento = (fecha, estado) => {
    const base = format(fecha, "dd/MM/yyyy 'a las' HH:mm");
    const estados = {
        pendiente: `Programado para el ${base}`,
        activo: `En curso desde el ${base}`,
        completado: `Finalizado el ${base}`,
        cancelado: `Cancelado (era el ${base})`
    };
    return estados[estado] || base;
};

const formatoUrgencia = (fecha) => {

    } else if (horasRestantes < 24) {
        return format(fecha, "'Hoy a las' HH:mm");
    } else if (horasRestantes < 48) {
        return format(fecha, "'Mañana a las' HH:mm");
    } else {
        return format(fecha, "dd/MM 'a las' HH:mm");
    }
};

// Uso en componentes React/Vue
const ComponenteFechaInteligente = ({ fecha, contexto = 'detalle' }) => {
    return formatosContextuales[contexto]?.(fecha) || format(fecha, 'PPP');
};

2. Funciones puras

Determinismo

Iguales entradas, iguales salidas. Facilita depuración y pruebas.

Sin efectos secundarios

No modifica datos externos ni depende de ellos. Trabaja con parámetros y valores retornados.

Composición

Funciona como bloques LEGO: se pueden encadenar para producir transformaciones complejas.

Relación directa con la inmutabilidad: se evitan datos cambiantes para mantener la pureza.

Pureza en código

// Función pura
const aplicarDescuento = (precio, porcentaje) => precio * (1 - porcentaje);

// Función impura
let totalCarrito = 0;
const agregarAlCarrito = (precio) => {
  totalCarrito += precio; // muta estado externo
  return totalCarrito;
};

La pureza depende de no leer ni mutar estados externos y de usar datos inmutables.

Checklist rápido

  • ¿Depende solo de sus argumentos?
  • ¿Evita modificar estructuras existentes y usa copias?
  • ¿Retorna siempre el mismo valor para la misma entrada?
Cuando la respuesta es “no”, encapsula la parte con efectos en otra función y deja este núcleo puro.

Recursión controlada

Clave: Define caso base claro y evita recursión infinita; usa memoización cuando el problema se vuelve exponencial. 
const fibonacci = (n, cache = {}) => {
  if (n < 2) return 1;
  if (cache[n]) return cache[n];
  cache[n] = fibonacci(n - 1, cache) + fibonacci(n - 2, cache);
  return cache[n];
};

Este patrón contrasta la elegancia declarativa con la necesidad de controlar recursos.

Pipelines declarativos

const ventas = [
  { vendedor: "Ana", total: 1200, estado: "cerrado" },
  { vendedor: "Luis", total: 400, estado: "abierto" },
  { vendedor: "Ana", total: 800, estado: "cerrado" },
  { vendedor: "María", total: 600, estado: "cerrado" }
];

const totalCerradoPorVendedor = (nombre) =>
  ventas
    .filter((venta) => venta.estado === "cerrado" && venta.vendedor === nombre)
    .map((venta) => venta.total)
    .reduce((acc, total) => acc + total, 0);

Practica ajustando el pipeline para aplicar impuestos, descuentos o métricas adicionales.
